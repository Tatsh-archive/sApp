1 /**
  2  * Represents an overlay view.
  3  * @constructor
  4  * @returns {sOverlayView} The view.
  5  */
  6 var sOverlayView = function () {
  7   this.parent.constructor.call(this);
  8 
  9   /**
 10    * @type Element
 11    * @private
 12    */
 13   this._DOMElement = document.createElement('div');
 14   this._DOMElement.className = 'soverlay';
 15 
 16   /**
 17    * @type sOverlayView
 18    * @private
 19    */
 20   var instance = this;
 21 
 22   /**
 23    * Default opacity.
 24    * @private
 25    * @type number
 26    */
 27   this._opacity = 0.6;
 28 
 29   /**
 30    * @type Object
 31    * @private
 32    */
 33   var css = {
 34     'display': 'none',
 35     'zIndex': -2000,
 36     'width': sWin.getWidth() + 'px',
 37     'position': 'fixed',
 38     'top': 0,
 39     'height': sWin.getHeight() + 'px',
 40     'opacity': 0,
 41     'filter': 'alpha(opacity=0)'
 42   };
 43   for (var attr in css) {
 44     this._DOMElement.style[attr] = css[attr];
 45   }
 46 
 47   sWin.addEventListener('resize', function () {
 48     var el = instance.getDOMElement();
 49     el.style.height = sWin.getHeight() + 'px';
 50     el.style.width = sWin.getWidth() + 'px';
 51   });
 52 
 53   return this;
 54 };
 55 /**
 56  * @type sView
 57  * @private
 58  */
 59 sOverlayView.prototype = new sView();
 60 sOverlayView.prototype.parent = sView.prototype;
 61 /**
 62  * Show the overlay.
 63  * @returns {sOverlayView} The object to allow method chaining.
 64  */
 65 sOverlayView.prototype.show = function () {
 66   this._DOMElement.style.filter = 'alpha(opacity=60)';
 67   this._DOMElement.style.display = 'block';
 68   this._DOMElement.style.zIndex = 100;
 69   this._DOMElement.style.opacity = this._opacity;
 70   return this;
 71 };
 72 /**
 73  * Hide the overlay.
 74  * @returns {sOverlayView} The object to allow method chaining.
 75  */
 76 sOverlayView.prototype.hide = function () {
 77   this._DOMElement.style.filter = 'alpha(opacity=0)';
 78   this._DOMElement.style.opacity = 0;
 79   this._DOMElement.style.zIndex = -2000;
 80   this._DOMElement.style.display = 'none';
 81   return this;
 82 };
 83 /**
 84  * @type sOverlayView
 85  * @private
 86  */
 87 sOverlayView._static = new sOverlayView();
 88 /**
 89  * Get the singleton instance.
 90  * @returns {sOverlayView} The object.
 91  */
 92 sOverlayView.getInstance = function () {
 93   if (!sOverlayView._static.isAppended()) {
 94     sOverlayView._static.appendTo(document.body);
 95   }
 96   return sOverlayView._static;
 97 };
