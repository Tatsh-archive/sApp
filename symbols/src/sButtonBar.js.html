1 /**
  2  * Represents a set of buttons.
  3  * @constructor
  4  * @returns {sButtonBar} The button bar object.
  5  */
  6 var sButtonBar = function () {
  7   this.parent.constructor.call(this);
  8 
  9   /**
 10    * @type Element
 11    * @private
 12    */
 13   this._DOMElement = sDoc.newElement('ul');
 14   this._DOMElement.className = 'sbutton-bar';
 15 
 16   /**
 17    * @type Object
 18    * @private
 19    */
 20   this._childElements = {};
 21 
 22   /**
 23    * @type Object
 24    * @private
 25    */
 26   this._eventHandlers = {};
 27 
 28   return this;
 29 };
 30 /**
 31  * @type sView
 32  * @private
 33  */
 34 sButtonBar.prototype = new sView();
 35 sButtonBar.prototype.parent = sView.prototype;
 36 /**
 37  * Adds a button.
 38  * @param {string} label Label text.
 39  * @param {function()} fn Callback.
 40  * @returns {sButtonBar} The object to allow method chaining.
 41  */
 42 sButtonBar.prototype.addButton = function (label, fn) {
 43   var eventHandler = function (event) {
 44     fn();
 45     event.preventDefault();
 46     return false;
 47   };
 48 
 49   var li = q(sDoc.newElement('li'));
 50   var realLi = li.get();
 51   var a = q(sDoc.newElement('a')).setAttributes({
 52     'href': '#'
 53   }).bind('click', eventHandler).setText(label);
 54 
 55   li.append(a);
 56 
 57   this._childElements[label] = realLi;
 58   this._eventHandlers[label] = eventHandler;
 59   this._DOMElement.appendChild(realLi);
 60 
 61   return this;
 62 };
 63 /**
 64  * Removes a button.
 65  * @param {string} label Label of the button to remove.
 66  * @returns {sButtonBar} The object to allow method chaining.
 67  */
 68 sButtonBar.prototype.removeButton = function (label) {
 69   if (this._childElements[label]) {
 70     var eventHandler = this._eventHandlers[label];
 71     q(this._childElements[label]).unbind('click', eventHandler).remove();
 72   }
 73 
 74   return this;
 75 };
