1 /**
  2  * Represents a link (anchor tag) with an image.
  3  * @constructor
  4  * @param {string} url URL for the link.
  5  * @param {string} src Source URI for the image.
  6  * @param {string} [alt] Alternative text for the anchor tag.
  7  * @param {string} [title] Title for the anchor tag.
  8  * @returns {sImageLink} The image link object.
  9  * @augments sView
 10  */
 11 var sImageLink = function (url, src, alt, title) {
 12   this.parent.constructor.call(this);
 13 
 14   /**
 15    * @type sImage
 16    * @private
 17    */
 18   this._image = new sImage(src, alt);
 19 
 20   /**
 21    * @type string
 22    * @private
 23    */
 24   this._title = '';
 25 
 26   /**
 27    * @type string
 28    * @private
 29    */
 30   this._url = url;
 31 
 32   /**
 33    * @type (Element|null)
 34    * @private
 35    */
 36   this._DOMElement = null;
 37 
 38   this.render();
 39 
 40   return this;
 41 };
 42 /**
 43  * @type sView
 44  * @private
 45  */
 46 sImageLink.prototype = new sView();
 47 sImageLink.prototype.parent = sView.prototype;
 48 /**
 49  * Renders the view.
 50  * @param {boolean} [force=false] Force re-rendering.
 51  * @returns {sImageLink} The object to allow method chaining.
 52  */
 53 sImageLink.prototype.render = function (force) {
 54   if (!this.isRendered() || !this._DOMElement || force) {
 55     if (!this._DOMElement) {
 56       this._DOMElement = sDoc.newElement('a');
 57     }
 58 
 59     q(this._DOMElement).removeChildren();
 60     this._DOMElement.className = 'hvideo-list-image';
 61 
 62     this._DOMElement.setAttribute('href', this._url);
 63     this._image.appendTo(this._DOMElement);
 64 
 65     this.setRendered(true);
 66   }
 67 
 68   return this;
 69 };
 70 /**
 71  * Gets the image object.
 72  * @returns {sImage} The image object.
 73  */
 74 sImageLink.prototype.getImage = function () {
 75   return this._image;
 76 };
 77 /**
 78  * Sets the title for the a tag.
 79  * @param {string} title The title.
 80  * @returns {sImageLink} The object to allow method chaining.
 81  */
 82 sImageLink.prototype.setTitle = function (title) {
 83   this._title = title;
 84   this._DOMElement.setAttribute('title', title);
 85   return this;
 86 };
 87 /**
 88  * Gets the title of the a tag.
 89  * @returns {string} The title.
 90  */
 91 sImageLink.prototype.getTitle = function (title) {
 92   return this._title;
 93 };
 94 /**
 95  * Set the image for this link.
 96  * @param {sImage} image The image object to use.
 97  * @returns {sImageLink} The object to allow method chaining.
 98  */
 99 sImageLink.prototype.setImage = function (image) {
100   this._image = image;
101 
102   q(this._DOMElement).removeChildren();
103 
104   this._DOMElement.appendChild(this._image.getDOMElement());
105 
106   return this;
107 };
