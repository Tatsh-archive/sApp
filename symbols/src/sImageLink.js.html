1 /**
  2  * Represents a link with an image.
  3  * @constructor
  4  * @param {string} url URL for the link.
  5  * @param {string} src Source URI for the image.
  6  * @param {string} alt Alternative txt for the a tag.
  7  * @returns {sImageLink} The image link object.
  8  * @augments sView
  9  */
 10 var sImageLink = function (url, src, alt) {
 11   this.parent.constructor.call(this);
 12 
 13   this._image = new sImage(src, alt);
 14 
 15   this._title = '';
 16   this._DOMElement = document.createElement('a');
 17   this._DOMElement.setAttribute('href', url);
 18   this._image.appendTo(this._DOMElement);
 19 
 20   return this;
 21 };
 22 /**
 23  * @type sView
 24  * @private
 25  */
 26 sImageLink.prototype = new sView();
 27 sImageLink.prototype.parent = sView.prototype;
 28 /**
 29  * Gets the image object.
 30  * @returns {sImage} The image object.
 31  */
 32 sImageLink.prototype.getImage = function () {
 33   return this._image;
 34 };
 35 /**
 36  * Sets the title for the a tag.
 37  * @param {string} title The title.
 38  * @returns {sImageLink} The object to allow method chaining.
 39  */
 40 sImageLink.prototype.setTitle = function (title) {
 41   this._title = title;
 42   this._DOMElement.setAttribute('title', title);
 43   return this;
 44 };
 45 /**
 46  * Gets the title of the a tag.
 47  * @returns {string} The title.
 48  */
 49 sImageLink.prototype.getTitle = function (title) {
 50   return this._title;
 51 };
 52 /**
 53  * Set the image for this link.
 54  * @param {sImage} image The image object to use.
 55  * @returns {sImageLink} The object to allow method chaining.
 56  */
 57 sImageLink.prototype.setImage = function (image) {
 58   this._image = image;
 59   this._DOMElement.innerHTML = '';
 60   this._DOMElement.appendChild(this._image.getDOMElement());
 61   return this;
 62 };
