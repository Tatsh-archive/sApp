1 /**
  2  * Represents a dialog, which typically appear in the middle of the window.
  3  * @constructor
  4  * @param {string} title The title for the dialog.
  5  * @returns {sDialog} The dialog object.
  6  * @augments sView
  7  */
  8 var sDialog = function (title) {
  9   // Just in case, must come before any others to make it easier to use
 10   var overlay = sOverlayView.getInstance();
 11 
 12   q(overlay.getDOMElement()).bind('click', function () {
 13     sHistory.removeState('dialog');
 14   });
 15 
 16   this.parent.constructor.call(this);
 17 
 18   /**
 19    * @type string
 20    * @private
 21    */
 22   this._title = title;
 23 
 24   /**
 25    * @type Element
 26    * @private
 27    */
 28   this._DOMElement = document.createElement('div');
 29   this._DOMElement.className = 'sdialog';
 30 
 31   /**
 32    * @type Element
 33    * @private
 34    */
 35   this._header = document.createElement('header');
 36   var h2 = document.createElement('h2');
 37   h2.appendChild(document.createTextNode(title));
 38   this._header.appendChild(h2);
 39   this._header.appendChild(sDialog.makeCloseButton());
 40   this._DOMElement.appendChild(this._header);
 41 
 42   /**
 43    * @type (Element|null)
 44    * @private
 45    */
 46   this._form = null;
 47 
 48   /**
 49    * @type Array
 50    * @private
 51    */
 52   this._buttons = [];
 53 
 54   /**
 55    * @type Element
 56    * @private
 57    */
 58   this._buttonsContainer = document.createElement('div');
 59   this._buttonsContainer.className = 'sdialog-button-container';
 60 
 61   /**
 62    * @type boolean
 63    * @private
 64    */
 65   this._centered = true;
 66 
 67   /**
 68    * @type (number|null)
 69    * @private
 70    */
 71   this._minYOnResize = null;
 72 
 73   return this;
 74 };
 75 /**
 76  * @private
 77  * @returns {Element}
 78  */
 79 sDialog.makeCloseButton = function () {
 80   var close = document.createElement('a');
 81   close.className = 'close-button-x';
 82   close.setAttribute('href', '#');
 83   close.innerHTML = 'X';
 84 
 85   q(close).bind('click', function (event) {
 86     sHistory.removeState('dialog');
 87     event.preventDefault();
 88     return false;
 89   });
 90 
 91   return close;
 92 };

100 /**
101  * Set the form for this dialog. A dialog can only have one form.
102  * @param {sForm} form Form object.
103  * @returns {sDialog} The object to allow method chaining.
104  */
105 sDialog.prototype.setForm = function (form) {
106   this._DOMElement.innerHTML = '';
107   this._DOMElement.appendChild(this._header);
108   form.appendTo(this._DOMElement);
109   this._form = form;
110   return this;
111 };
112 /**
113  * Get the form for this dialog.
114  * @returns {sForm|null} The form object or null if one has not been set.
115  */
116 sDialog.prototype.getForm = function () {
117   return this._form;
118 };
119 /**
120  * Set the buttons for this form.
121  * @param {Array} buttons Set of DOM Elements of submit button type.
122  * @returns {sDialog} The object to allow method chaining.
123  */
124 sDialog.prototype.setButtons = function (buttons) {
125   this._buttons = [];
126   return this;
127 };

158 /**
159  * Hide the dialog. This adds a class <code>sdialog-hidden</code> to the
160  *   element.
161  * @returns {sDialog} The object to allow method chaining.
162  */
163 sDialog.prototype.hide = function () {
164   sOverlayView.getInstance().hide();
165   this._DOMElement.className += ' sdialog-hidden';
166   return this;
167 };
168 /**
169  * Show the dialog. This removes the class <code>sdialog-hidden</code> from
170  *   the main element.
171  * @returns {sDialog} The object to allow method chaining.
172  */
173 sDialog.prototype.show = function () {
174   sOverlayView.getInstance().show();
175   this._DOMElement.className = this._DOMElement.className.replace(/(\s+)?sdialog\-hidden/g, '');
176   return this;
177 };
178 /**
179  * Add a button.
180  * @param {string} label Label text for the button.
181  * @param {sForm} formToSubmit The form that this button will submit.
182  * @returns {sDialog} The object to allow method chaining.
183  */
184 sDialog.prototype.addButton = function (label, formToSubmit) {
185   var button = document.createElement('input');
186   button.setAttribute('type', 'submit');
187   button.setAttribute('value', label);
188   button.className = 'form-submit';
189 
190   q(button).bind('click', function (event) {
191     formToSubmit.getDOMElement().submit();
192   });
193 
194   this._buttons.push(button);
195 
196   return this;
197 };
198 /**
199  * Control the minimum y value the dialog must stay at when the window
200  *   gets resized.
201  * @param {number} y Y value.
202  * @returns {sDialog} The object to allow method chaining.
203  */
204 sDialog.prototype.setMinYOnResize = function (y) {
205   this._minYOnResize = y;
206   return this;
207 };
208 /**
209  * Append the dialog to the element specified.
210  * @param {Element} element Element to append to.
211  * @returns {sDialog} The object to allow method chaining.
212  */
213 sDialog.prototype.appendTo = function (element) {
214   var addEvent = !this.isAppended() && this._centered;
215 
216   this.parent.appendTo.call(this, element);
217 
218   for (var i = 0; i < this._buttons.length; i++) {
219     this._buttonsContainer.appendChild(this._buttons[i]);
220   }
221 
222   this._DOMElement.appendChild(this._buttonsContainer);
223 
224   if (addEvent) {
225     element = this._DOMElement;
226     var minY = this._minYOnResize;
227     var resize = function () {
228       var height = sWin.getHeight() / 2;
229       var width = sWin.getWidth() / 2;
230       var x = parseInt(width - (element.offsetWidth / 2), 10);
231       var y = parseInt(height - (element.offsetHeight / 2), 10);
232 
233       if (minY !== null && y < minY) {
234         y = minY;
235       }
236 
237       if (sCSS.hasTransforms) {
238         sCSS.translate(element, x, y);
239       }
240       else {
241         element.style.left = x + 'px';
242         element.style.top = y + 'px';
243       }
244     };
245     resize();
246     sWin.bind('resize', resize, false);
247   }
248 
249   return this;
250 };
251 /**
252  * Set the width of the main element.
253  * @param {number} width The width to set.
254  * @param {string} [unit=px] The unit to use.
255  * @returns {sDialog} The object to allow method chaining.
256  */
257 sDialog.prototype.setWidth = function (width, unit) {
258   if (unit === undefined) {
259     unit = 'px';
260   }
261   this._DOMElement.style.width = width + unit;
262   return this;
263 };
