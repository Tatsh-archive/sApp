1 /**
  2  * Represents a dialog, which typically appear in the middle of the window.
  3  * @constructor
  4  * @param {string} title The title for the dialog.
  5  * @returns {sDialog} The dialog object.
  6  * @augments sView
  7  */
  8 var sDialog = function (title) {
  9   this.parent.constructor.call(this);
 10 
 11   /**
 12    * @type string
 13    * @private
 14    */
 15   this._title = title;
 16 
 17   /**
 18    * @type Element
 19    * @private
 20    */
 21   this._DOMElement = document.createElement('div');
 22 
 23   /**
 24    * @type Element
 25    * @private
 26    */
 27   this._header = document.createElement('header');
 28 
 29   /**
 30    * @type (Element|null)
 31    * @private
 32    */
 33   this._form = null;
 34 
 35   /**
 36    * @type Array
 37    * @private
 38    */
 39   this._buttons = [];
 40 
 41   /**
 42    * @type Element
 43    * @private
 44    */
 45   this._buttonsContainer = document.createElement('div');
 46   this._buttonsContainer.className = 'sdialog-button-container';
 47 
 48   /**
 49    * @type boolean
 50    * @private
 51    */
 52   this._centered = true;
 53 
 54   /**
 55    * @type (number|null)
 56    * @private
 57    */
 58   this._minYOnResize = null;
 59 
 60   this._renderHeader();
 61 
 62   return this;
 63 };
 64 /**
 65  * @private
 66  * @returns {Element}
 67  */
 68 sDialog.makeCloseButton = function () {
 69   var close = document.createElement('a');
 70   close.className = 'close-button-x';
 71   close.setAttribute('href', '#');
 72   close.innerHTML = 'X';
 73 
 74   q(close).bind('click', function (event) {
 75     sHistory.removeState('dialog');
 76     event.preventDefault();
 77     return false;
 78   });
 79 
 80   return close;
 81 };

 88 sDialog.prototype.constructor = sDialog;
 89 /**
 90  * @private
 91  * @returns {sDialog} The object to allow method chaining.
 92  */
 93 sDialog.prototype._renderHeader = function () {
 94   this._DOMElement.innerHTML = '';
 95   this._header.innerHTML = '';
 96 
 97   /**
 98    * @type Element
 99    * @private
100    */
101   var h2 = document.createElement('h2');
102 
103   this._DOMElement.className = 'sdialog';
104   this._header.className = 'sdialog-header';
105   h2.appendChild(document.createTextNode(this._title));
106   this._header.appendChild(h2);
107   this._header.appendChild(sDialog.makeCloseButton());
108   this._DOMElement.appendChild(this._header);
109 
110   return this;
111 };
112 /**
113  * Set the form for this dialog. A dialog can only have one form.
114  * @param {sForm} form Form object.
115  * @returns {sDialog} The object to allow method chaining.
116  */
117 sDialog.prototype.setForm = function (form) {
118   this._renderHeader();
119   form.appendTo(this._DOMElement);
120   this._form = form;
121   return this;
122 };

179 /**

189 /**
190  * Add a button.
191  * @param {string} label Label text for the button.
192  * @param {sForm} formToSubmit The form that this button will submit.
193  * @returns {sDialog} The object to allow method chaining.
194  */
195 sDialog.prototype.addButton = function (label, formToSubmit) {
196   var button = document.createElement('input');
197   button.setAttribute('type', 'submit');
198   button.setAttribute('value', label);
199   button.className = 'form-submit';
200 
201   q(button).bind('click', function (event) {
202     formToSubmit.getDOMElement().submit();
203   });
204 
205   this._buttons.push(button);
206 
207   return this;
208 };
209 /**
210  * Control the minimum y value the dialog must stay at when the window
211  *   gets resized.
212  * @param {number} y Y value.
213  * @returns {sDialog} The object to allow method chaining.
214  */
215 sDialog.prototype.setMinYOnResize = function (y) {
216   this._minYOnResize = y;
217   return this;
218 };
219 /**
220  * Append the dialog to the element specified.
221  * @param {Element} element Element to append to.
222  * @returns {sDialog} The object to allow method chaining.
223  */
224 sDialog.prototype.appendTo = function (element) {
225   var addEvent = !this.isAppended() && this._centered;
226 
227   this.parent.appendTo.call(this, element);
228 
229   for (var i = 0; i < this._buttons.length; i++) {
230     this._buttonsContainer.appendChild(this._buttons[i]);
231   }
232 
233   this._DOMElement.appendChild(this._buttonsContainer);
234 
235   if (addEvent) {
236     element = this._DOMElement;
237     var minY = this._minYOnResize;
238     var resize = function () {
239       var height = sWin.getHeight() / 2;
240       var width = sWin.getWidth() / 2;
241       var x = parseInt(width - (element.offsetWidth / 2), 10);
242       var y = parseInt(height - (element.offsetHeight / 2), 10);
243 
244       if (minY !== null && y < minY) {
245         y = minY;
246       }
247 
248       if (sCSS.hasTransforms) {
249         sCSS.translate(element, x, y);
250       }
251       else {
252         element.style.left = x + 'px';
253         element.style.top = y + 'px';
254       }
255     };
256     resize();
257     sWin.bind('resize', resize, false);
258   }
259 
260   return this;
261 };
262 /**
263  * Set the width of the main element.
264  * @param {number} width The width to set.
265  * @param {string} [unit=px] The unit to use.
266  * @returns {sDialog} The object to allow method chaining.
267  */
268 sDialog.prototype.setWidth = function (width, unit) {
269   if (unit === undefined) {
270     unit = 'px';
271   }
272   this._DOMElement.style.width = width + unit;
273   return this;
274 };
