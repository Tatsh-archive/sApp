1 /**
  2  * Base class for form fields.
  3  * @constructor
  4  * @returns {sFormField} The form field object.
  5  * @augments sView
  6  */
  7 var sFormField = function () {
  8   this.parent.constructor.call(this);
  9 
 10   /**
 11    * @type Element
 12    * @private
 13    */
 14   this._fieldDOMElement = document.createElement('input');
 15 
 16   /**
 17    * @type string
 18    * @private
 19    */
 20   this._value = '';
 21 
 22   /**
 23    * The main DOM element.
 24    * @private
 25    * @type Element
 26    */
 27   this._DOMElement = document.createElement('div');
 28   this._DOMElement.appendChild(this._fieldDOMElement);
 29 
 30   return this;
 31 };
 32 /**
 33  * @type sView
 34  * @private
 35  */
 36 sFormField.prototype = new sView();
 37 sFormField.prototype.parent = sView.prototype;
 38 /**
 39  * Get the field DOM element, since fields are wrapped in divs.
 40  * @returns {Element} The DOM element.
 41  */
 42 sFormField.prototype.getFieldDOMElement = function () {
 43   return this._fieldDOMElement;
 44 };
 45 /**
 46  * Set the field's required attribute.
 47  * @param {boolean} required If required.
 48  * @returns {sFormField} The object to allow method chaining.
 49  */
 50 sFormField.prototype.setRequired = function (required) {
 51   this._fieldDOMElement.required = required ? true : false;
 52   return this;
 53 };
