1 /**
  2  * This class is not intended for general usage. It is intended for
  3  *   extension.
  4  * When extending, before sure to override the decodeDataToItems() method.
  5  * @see sDataList#decodeDataToItems
  6  * @constructor
  7  * @returns {sDataList} The object.
  8  * @augments sView
  9  */
 10 var sDataList = function () {
 11   /**
 12    * @private
 13    * @type (string|null)
 14    */
 15   this._dataSourceURI = null;
 16   /**
 17    * @private
 18    * @type Array
 19    */
 20   this._data = [];
 21   /**
 22    * @private
 23    * @type number
 24    */
 25   this._pageNumber = 1;
 26   /**
 27    * @private
 28    * @type number
 29    */
 30   this._limitPerRequest = 20;
 31   /**
 32    * @private
 33    * @type string
 34    */
 35   this._pageParameter = 'page';
 36   /**
 37    * @private
 38    * @type string
 39    */
 40   this._limitParameter = 'limit';
 41   /**
 42    * @private
 43    * @type Element
 44    */
 45   this._DOMElement = sDoc.newElement('ul');
 46   /**
 47    * @private
 48    * @type boolean
 49    */
 50   this._rendered = false;
 51 
 52   return this;
 53 };
 54 /**
 55  * @type sView
 56  * @private
 57  */
 58 sDataList.prototype = new sView();
 59 /**
 60  * Set the data source.
 61  * @param {string} url URL to the data source, relative or full.
 62  * @returns {sDataList} The data list object to allow method chaining.
 63  */
 64 sDataList.prototype.setDataSourceURI = function (url) {
 65   this._dataSourceURI = url;
 66   return this;
 67 };
 68 /**
 69  * Set the page parameter.
 70  * @param {string} param The parameter name to use to name the page number
 71  *   in each request.
 72  * @return {sDataList} The data list object to allow method chaining.
 73  */
 74 sDataList.prototype.setDataSourceURIPageParameter = function (param) {
 75   this._pageParameter = param;
 76   return this;
 77 };
 78 /**
 79  * Set the limit parameter.
 80  * @param {string} param The parameter name to use to limit the count of items
 81  *   received in each request.
 82  * @return {sDataList} The data list object to allow method chaining.
 83  */
 84 sDataList.prototype.setDataSourceURILimitParameter = function (param) {
 85   this._limitParameter = param;
 86   return this;
 87 };

104 /**
105  * Set the limit of items per request.
106  * @param {number} limit Limit.
107  * @returns {sDataList} The data list object to allow method chaining.
108  */
109 sDataList.prototype.setLimitPerRequest = function (limit) {
110   this._limitPerRequest = parseInt(limit, 10);
111   return this;
112 };
113 /**
114  * Gets the number of items per request.
115  * @returns {number} Number of items per request.
116  */
117 sDataList.prototype.getLimitPerRequest = function () {
118   return this._limitPerRequest;
119 };
120 /**
121  * Set the page number.
122  * @param {number} page Page number.
123  * @returns {sDataList} The data list object to allow method chaining.
124  */
125 sDataList.prototype.setPageNumber = function (page) {
126   this._pageNumber = page;
127   return this;
128 };
129 /**
130  * Set the data. It may be desirable to re-render after calling this.
131  * @param {Array} data The data to set.
132  * @returns {sDataList} The data list object to allow method chaining.
133  */
134 sDataList.prototype.setData = function (data) {
135   this._data = data;
136   return this;
137 };
138 /**
139  * Renders the data. When implementing, be sure to call setRendered with true
140  *   when rendering is complete.
141  * @param {boolean} force Force a re-load of the data.
142  * @returns {sDataList} The data list object to allow method chaining.
143  */
144 sDataList.prototype.render = function (force) {
145   if (!this._rendered || force) {
146     this._rendered = true;
147   }
148   return this;
149 };
150 /**
151  * Returns if rendered.
152  * @returns {boolean} If the list is already rendered.
153  */
154 sDataList.prototype.isRendered = function () {
155   return this._rendered;
156 };
157 /**
158  * Set rendered state.
159  * @param {boolean} state True or false.
160  * @returns {sDataList} The data list object to allow method chaining.
161  */
162 sDataList.prototype.setRendered = function (state) {
163   this._rendered = state ? true : false;
164   return this;
165 };
166 /**
167  * Gets the data.
168  * @returns {Array} The data last retrieived, or empty array.
169  */
170 sDataList.prototype.getData = function () {
171   return this._data;
172 };
