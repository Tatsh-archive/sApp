1 /**
  2  * This class is not intended for general usage. It is intended for
  3  *   extension.
  4  * When extending, before sure to override the decodeDataToItems() method.
  5  * @see sDataList#decodeDataToItems
  6  * @constructor
  7  * @returns {sDataList} The object.
  8  * @augments sView
  9  */
 10 var sDataList = function () {
 11   /**
 12    * @private
 13    * @type (string|null)
 14    */
 15   this._dataSourceURI = null;
 16   /**
 17    * @private
 18    * @type Array
 19    */
 20   this._data = [];
 21   /**
 22    * @private
 23    * @type number
 24    */
 25   this._pageNumber = 1;
 26   /**
 27    * @private
 28    * @type number
 29    */
 30   this._limitPerRequest = 20;
 31   /**
 32    * @private
 33    * @type string
 34    */
 35   this._pageParameter = 'page';
 36   /**
 37    * @private
 38    * @type string
 39    */
 40   this._limitParameter = 'limit';
 41   /**
 42    * @private
 43    * @type Element
 44    */
 45   this._mainDOMElement = sDoc.newElement('div');
 46   /**
 47    * @private
 48    * @type boolean
 49    */
 50   this._rendered = false;
 51   /**
 52    * @type sSortableButtonBar
 53    * @private
 54    */
 55   this._sortableButtonBar = new sButtonBar();
 56   /**
 57    * @type (string|null)
 58    * @private
 59    */
 60   this._currentSort = null;
 61 
 62   this._mainDOMElement.appendChild(this._sortableButtonBar.getDOMElement());
 63 
 64   return this;
 65 };

 67  * @type sView

164 /**
165  * Set the data. It may be desirable to re-render after calling this.
166  * @param {Array} data The data to set.
167  * @returns {sDataList} The data list object to allow method chaining.
168  */
169 sDataList.prototype.setData = function (data) {
170   this._data = data;
171   return this;
172 };
173 /**
174  * Renders the data. When implementing, be sure to call setRendered with true
175  *   when rendering is complete.
176  * @param {boolean} force Force a re-load of the data.
177  * @returns {sDataList} The data list object to allow method chaining.
178  */
179 sDataList.prototype.render = function (force) {
180   if (!this._rendered || force) {
181     this._rendered = true;
182   }
183   return this;
184 };
185 /**
186  * Returns if rendered.
187  * @returns {boolean} If the list is already rendered.
188  */
189 sDataList.prototype.isRendered = function () {
190   return this._rendered;
191 };
192 /**
193  * Set rendered state.
194  * @param {boolean} state True or false.
195  * @returns {sDataList} The data list object to allow method chaining.
196  */
197 sDataList.prototype.setRendered = function (state) {
198   this._rendered = state ? true : false;
199   return this;
200 };

208 /**
209  * Adds a category button to sort with.
210  * @param {string} categoryName Category name, such as <code>'A-Z'</code>.
211  * @param {function(string)} fn Function that will be called when this is sort
212  *   is requested.
213  * @returns {sDataList} The data list object to allow method chaining.
214  */
215 sDataList.prototype.addSortButton = function (label, fn) {
216   var instance = this;
217   this._sortableButtonBar.addButton(label, function () {
218     fn();
219   });
220   return this;
221 };
222 /**
223  * Appends the main element to the element specified.
224  * @param {Element} element The element to append to.
225  * @returns {sDataList} The object to allow method chaining.
226  */
227 sDataList.prototype.appendTo = function (element) {
228   this._isAppended = true;
229   this._rendered = true;
230   element.appendChild(this._mainDOMElement);
231   return this;
232 };
