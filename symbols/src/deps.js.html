1 /**
  2  * Adds placeholder support to older browsers. Does not support password fields
  3  *   yet.
  4  * @param {Element} input Input or textarea element.
  5  * @param {string} [color='#aaa'] Color string. Recommend hash format unless
  6  *   detecting if the browser has support for rgb/rgba/hsl/hsla/other.
  7  * @return {Element} The element passed in.
  8  * @private
  9  */
 10 var inputPlaceholder = function (input, color) {
 11   if (color && color.toString) {
 12     color = color.toString();
 13   }
 14   else if (typeof color !== 'string') {
 15     color = '#aaa';
 16   }
 17 
 18   /**
 19    * Tested with:
 20    * - IE 7
 21    * - IE 8
 22    * - IE 9
 23    * - Firefox 3.6
 24    */
 25 
 26   // No support for password yet
 27   var type = input.getAttribute('type');
 28   if (type !== null && type.toLowerCase() === 'password') {
 29     return input;
 30   }
 31 
 32   // Do nothing if placeholder supported by the browser (Webkit, Firefox 3.7)
 33   if (input.placeholder && 'placeholder' in document.createElement(input.tagName)) {
 34     return input;
 35   }
 36 
 37   var defaultColor = input.style.color;
 38   var placeholder = input.getAttribute('placeholder');
 39   var addEvent = 'addEventListener';
 40   var event = 'focus';
 41   var blurEvent = 'blur';
 42   var submitEvent = 'submit';
 43 
 44   if (input.value === '' || input.value == placeholder) {
 45     input.value = placeholder;
 46     input.style.color = color;
 47     input.setAttribute('data-placeholder-visible', true);
 48   }
 49 
 50   if (input.attachEvent) {
 51     addEvent = 'attachEvent';
 52     event = 'onfocus';
 53     blurEvent = 'onblur';
 54     submitEvent = 'onsubmit';
 55   }
 56 
 57   input[addEvent](event, function() {
 58     input.style.color = defaultColor;
 59 
 60     if (input.getAttribute('data-placeholder-visible')) {
 61       input.removeAttribute('data-placeholder-visible');
 62       input.value = '';
 63     }
 64   }, false);
 65 
 66   input[addEvent](blurEvent, function() {
 67     if (input.value === '') {
 68       input.setAttribute('data-placeholder-visible', true);
 69       input.value = placeholder;
 70       input.style.color = color;
 71     } else {
 72       input.style.color = defaultColor;
 73       input.removeAttribute('data-placeholder-visible');
 74     }
 75   }, false);
 76 
 77   if (input.form) {
 78     input.form[addEvent](submitEvent, function() {
 79       if (input.getAttribute('data-placeholder-visible')) {
 80         input.value = '';
 81       }
 82     }, false);
 83   }
 84 
 85   return input;
 86 };
