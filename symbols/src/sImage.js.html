1 /**
  2  * Represents an image.
  3  * @constructor
  4  * @param {string} [src] Source of the image.
  5  * @param {string} [alt] Alternative text for the image.
  6  * @returns {sImage} The image object.
  7  * @augments sView
  8  */
  9 var sImage = function (src, alt) {
 10   this.parent.constructor.call(this);
 11 
 12   if (alt === undefined) {
 13     alt = '';
 14   }
 15   if (src === undefined) {
 16     src = '';
 17   }
 18 
 19   /**
 20    * @type string
 21    * @private
 22    */
 23   this._alt = alt;
 24 
 25   /**
 26    * @type string
 27    * @private
 28    */
 29   this._src = src;
 30 
 31   /**
 32    * @type number
 33    * @private
 34    */
 35   this._height = 0;
 36 
 37   /**
 38    * @type number
 39    * @private
 40    */
 41   this._width = 0;
 42 
 43   /**
 44    * @type (Element|null)
 45    * @private
 46    */
 47   this._DOMElement = null;
 48 
 49   if (this._src) {
 50     this.render();
 51   }
 52 
 53   return this;
 54 };
 55 /**
 56  * @type sView
 57  * @private
 58  */
 59 sImage.prototype = new sView();
 60 sImage.prototype.parent = sView.prototype;
 61 /**
 62  * Render the view.
 63  * @param {boolean} [force=false] Force the re-rendering.
 64  * @returns {sImage} The object to allow method chaining.
 65  */
 66 sImage.prototype.render = function (force) {
 67   if (!this.isRendered() || force) {
 68     this._DOMElement = document.createElement('img');
 69     this._DOMElement.setAttribute('src', this._src);
 70     this._DOMElement.setAttribute('alt', this._alt);
 71 
 72     if (this._width) {
 73       this._DOMElement.setAttribute('width', this._width);
 74     }
 75     if (this._height) {
 76       this._DOMElement.setAttribute('height', this._height);
 77     }
 78 
 79     this.setRendered(true);
 80   }
 81   return this;
 82 };
 83 /**
 84  * Set the source URI of the image.
 85  * @param {string} src The URI of the image.
 86  * @param {function()} [onloadHandler] The onload handler function.
 87  * @returns {sImage} The object to allow method chaining.
 88  */
 89 sImage.prototype.setSource = function (src, onloadHandler) {
 90   var sel = q(this._DOMElement);
 91 
 92   if (onloadHandler !== undefined) {
 93     sel.unbind('load', this._onloadHandler);
 94     this._onloadHandler = onloadHandler;
 95   }
 96   else {
 97     this._onloadHandler = function () {};
 98   }
 99 
100   if (this.isAppended()) {
101     sel.bind('load', this._onloadHandler).setAttributes({'src': src});
102     return this;
103   }
104 
105   this._DOMElement = document.createElement('img');
106   this._DOMElement.setAttribute('src', src);
107   this._DOMElement.setAttribute('alt', this._alt);
108 
109   return this;
110 };
111 /**
112  * Set the alternative text for the image.
113  * @param {string} alt The alternative text.
114  * @returns {sImage} The ojbect to allow method chaining.
115  */
116 sImage.prototype.setAlt = function (alt) {
117   this._alt = alt;
118   this._DOMElement.setAttribute('alt', alt);
119   return this;
120 };
121 /**
122  * Set the height of the DOM element.
123  * @param {number} height Height.
124  * @returns {sImage} The ojbect to allow method chaining.
125  */
126 sImage.prototype.setHeight = function (height) {
127   this._height = height;
128   this._DOMElement.setAttribute('height', height);
129   return this;
130 };
131 /**
132  * Set the width of the DOM element.
133  * @param {number} width Width.
134  * @returns {sImage} The ojbect to allow method chaining.
135  */
136 sImage.prototype.setWidth = function (width) {
137   this._width = width;
138   this._DOMElement.setAttribute('width', width);
139   return this;
140 };
