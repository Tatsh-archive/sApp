1 /**
  2  * Represents an image.
  3  * @constructor
  4  * @param {string} src Source of the image.
  5  * @param {string} alt Alternative text for the image.
  6  * @returns {sImage} The image object.
  7  * @augments sView
  8  */
  9 var sImage = function (src, alt) {
 10   this.parent.constructor.call(this);
 11 
 12   if (alt === undefined) {
 13     alt = '';
 14   }
 15 
 16   this._alt = alt;
 17   this._title = '';
 18   this._DOMElement = document.createElement('img');
 19   this._DOMElement.setAttribute('src', src);
 20   this._DOMElement.setAttribute('alt', this._alt);
 21 
 22   this._onloadHandler = function () {};
 23 
 24   return this;
 25 };
 26 /**
 27  * @type sView
 28  * @private
 29  */
 30 sImage.prototype = new sView();
 31 sImage.prototype.parent = sView.prototype;
 32 /**
 33  * Set the source URI of the image.
 34  * @param {string} src The URI of the image.
 35  * @param {function()} [onloadHandler] The onload handler function.
 36  * @returns {sImage} The ojbect to allow method chaining.
 37  */
 38 sImage.prototype.setSource = function (src, onloadHandler) {
 39   var sel = q(this._DOMElement);
 40 
 41   if (onloadHandler !== undefined) {
 42     sel.unbind('load', this._onloadHandler);
 43     this._onloadHandler = onloadHandler;
 44   }
 45   else {
 46     this._onloadHandler = function () {};
 47   }
 48 
 49   if (this.isAppended()) {
 50     sel.bind('load', this._onloadHandler).setAttributes({'src': src});
 51     return this;
 52   }
 53 
 54   this._DOMElement = document.createElement('img');
 55   this._DOMElement.setAttribute('src', src);
 56   this._DOMElement.setAttribute('alt', this._alt);
 57   return this;
 58 };
 59 /**
 60  * Set the alternative text for the image.
 61  * @param {string} alt The alternative text.
 62  * @returns {sImage} The ojbect to allow method chaining.
 63  */
 64 sImage.prototype.setAlt = function (alt) {
 65   this._alt = alt;
 66   this._DOMElement.setAttribute('alt', alt);
 67   return this;
 68 };
