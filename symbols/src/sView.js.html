1 /**
  2  * Base view class. All views inherit from this class.
  3  * @constructor
  4  * @returns {sView} The view object.
  5  */
  6 var sView = function () {
  7   /**
  8    * @type (Element|null)
  9    * @private
 10    */
 11   this._DOMElement = null;
 12   /**
 13    * @type boolean
 14    * @private
 15    */
 16   this._isAppended = false;
 17   /**
 18    * @type boolean
 19    * @private
 20    */
 21   this._rendered = false;
 22   return this;
 23 };
 24 /**
 25  * Set the DOM element. Resets render state.
 26  * @param {Element} element Element to set as this DOM element.
 27  * @returns {sView} The object to allow method chaining.
 28  */
 29 sView.prototype.setDOMElement = function (element) {
 30   this._DOMElement = element;
 31   this._rendered = false;
 32   return this;
 33 };
 34 /**
 35  * Get the DOM element.
 36  * @returns {Element} The DOM element.
 37  */
 38 sView.prototype.getDOMElement = function () {
 39   return this._DOMElement;
 40 };
 41 /**
 42  * If the DOM element is appended.
 43  * @returns {boolean} If the DOM element is appended.
 44  */
 45 sView.prototype.isAppended = function () {
 46   return this._isAppended;
 47 };
 48 /**
 49  * Appends the main element to the element specified.
 50  * @param {Element} element The element to append to.
 51  * @returns {sView} The object to allow method chaining.
 52  */
 53 sView.prototype.appendTo = function (element) {
 54   this._isAppended = true;
 55   this._rendered = true;
 56   element.appendChild(this._DOMElement);
 57   return this;
 58 };
 59 /**
 60  * Set the rendered state of this view.
 61  * @param {boolean} bool The value to set. True or false.
 62  * @returns {sView} The object to allow method chaining.
 63  */
 64 sView.prototype.setRendered = function (bool) {
 65   this._rendered = bool ? true : false;
 66   return this;
 67 };
 68 /**
 69  * If the view is rendered.
 70  * @returns {boolean} If the view is rendered.
 71  */
 72 sView.prototype.isRendered = function () {
 73   return this._rendered;
 74 };
 75 /**
 76  * Prepend the view to the element specified.
 77  * @param {Element} element The element to prepend to.
 78  * @returns {sView} The object to allow method chaining.
 79  */
 80 sView.prototype.prependTo = function (element) {
 81   this._isAppended = true;
 82   this._rendered = true;
 83   element.insertBefore(this._DOMElement, element.firstChild);
 84   return this;
 85 };
 86 /**
 87  * Insert the element after the specified ID.
 88  * @param {string} id ID to search for.
 89  * @returns {sView} The object to allow method chaining.
 90  */
 91 sView.prototype.insertAfterId = function (id) {
 92   var element = document.getElementById(id);
 93 
 94   if (element !== null) {
 95     if (element.nextSibling) {
 96       element.parentNode.insertBefore(this._DOMElement, element.nextSibling);
 97     }
 98     else {
 99       element.parentNode.appendChild(this._DOMElement);
100     }
101   }
102 
103   return this;
104 };
