1 /**
  2  * Represents a form password field.
  3  * @constructor
  4  * @param {string} name Name for the field.
  5  * @returns {sPasswordField} The password field object.
  6  * @augments sTextField
  7  */
  8 var sPasswordField = function (name) {
  9   this.parent.constructor.call(this, name);
 10 
 11   /**
 12    * @type Element
 13    * @private
 14    */
 15   this._DOMElement = document.createElement('div');
 16   this._DOMElement.className = 'form-password-container';
 17 
 18   /**
 19    * @type Element
 20    * @private
 21    */
 22   this._fieldDOMElement = document.createElement('input');
 23   this._fieldDOMElement.className = 'form-password';
 24   this._fieldDOMElement.setAttribute('type', 'password');
 25   this._fieldDOMElement.setAttribute('name', name);
 26 
 27   this._DOMElement.appendChild(this._fieldDOMElement);
 28 
 29   return this;
 30 };
 31 /**
 32  * @private
 33  * @type sTextField
 34  */
 35 sPasswordField.prototype = new sTextField('');
 36 sPasswordField.prototype.parent = sTextField.prototype;
