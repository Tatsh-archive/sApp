1 /**
  2  * Represents a form textarea field.
  3  * @constructor
  4  * @param {string} name Name for the field.
  5  * @returns {sTextAreaField} The textarea field object.
  6  * @augments sFormField
  7  */
  8 var sTextAreaField = function (name) {
  9   this.parent.constructor.call(this, name);
 10 
 11   /**
 12    * @type Element
 13    * @private
 14    */
 15   this._DOMElement = document.createElement('div');
 16   this._DOMElement.className = 'form-textarea-container';
 17 
 18   /**
 19    * @type Element
 20    * @private
 21    */
 22   this._fieldDOMElement = document.createElement('textarea');
 23   this._fieldDOMElement.setAttribute('name', name);
 24   this._fieldDOMElement.setAttribute('id', sHTML.makeFormElementID(name));
 25   this._fieldDOMElement.className = 'form-textarea';
 26 
 27   this._DOMElement.appendChild(this._fieldDOMElement);
 28 
 29   return this;
 30 };
 31 /**
 32  * @type sFormField
 33  * @private
 34  */
 35 sTextAreaField.prototype = new sFormField();
 36 sTextAreaField.prototype.parent = sFormField.prototype;
 37 sTextAreaField.prototype.constructor = sTextAreaField;
 38 /**
 39  * Set the value of the field.
 40  * @param {string} val The value.
 41  * @returns {sTextAreaField} The object to allow method chaining.
 42  */
 43 sTextAreaField.prototype.setValue = function (val) {
 44   this._fieldDOMElement.innerHTML = fHTML.decode(val);
 45   return this;
 46 };
 47 /**
 48  * Set the placeholder text of the field.
 49  * @param {string} placeholder The placeholder text.
 50  * @returns {sTextAreaField} The object to allow method chaining.
 51  */
 52 sTextAreaField.prototype.setPlaceholder = function (placeholder) {
 53   this._fieldDOMElement.setAttribute('placeholder', placeholder);
 54   inputPlaceholder(this._fieldDOMElement);
 55   return this;
 56 };
