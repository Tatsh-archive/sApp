1 /**
  2  * Represents a form.
  3  * @constructor
  4  * @returns {sForm} The form object.
  5  * @augments sView
  6  */
  7 var sForm = function () {
  8   this.parent.constructor.call(this);
  9 
 10   /**
 11    * @type Element
 12    * @private
 13    */
 14   this._DOMElement = document.createElement('form');
 15   this._DOMElement.className = 'sform';
 16 
 17   /**
 18    * @type Array
 19    * @private
 20    */
 21   this._fields = [];
 22   /**
 23    * @type Array
 24    * @private
 25    */
 26   this._submithandlers = [];
 27   /**
 28    * @type boolean
 29    * @private
 30    */
 31   this._usesDefaultEvent = true;
 32 
 33   /**
 34    * @type Array
 35    * @private
 36    */
 37   this._buttons = [];
 38 
 39   /**
 40    * @type Element
 41    * @private
 42    */
 43   this._buttonsDOMElement = document.createElement('div');
 44   this._buttonsDOMElement.className = 'form-ops-container';
 45 
 46   /**
 47    * @type Element
 48    * @private
 49    */
 50   this._CSRFDOMElement = document.createElement('input');
 51   this._CSRFDOMElement.setAttribute('type', 'hidden');
 52   this._CSRFDOMElement.setAttribute('name', 'csrf');
 53   this._CSRFDOMElement.value = '';
 54 
 55   /**
 56    * @type sForm
 57    * @private
 58    */
 59   var instance = this;
 60 
 61   q(this._DOMElement).bind('submit', function (event) {
 62     var fns = instance.getSubmitHandlers();
 63     var ret = true;
 64 
 65     for (var i = 0; i < fns.length; i++) {
 66       if (!fns[i](instance)) {
 67         ret = false;
 68       }
 69     }
 70 
 71     if (!instance.usesDefaultEvent() || !ret) {
 72       event.preventDefault();
 73       return false;
 74     }
 75   });
 76 
 77   return this;
 78 };

137 /**
138  * @private
139  */
140 sForm.prototype._pendCommon = function () {
141   this._isAppended = true;
142 
143   for (var i = 0; i < this._fields.length; i++) {
144     this._fields[i].appendTo(this._DOMElement);
145   }
146 
147   for (i = 0; i < this._buttons.length; i++) {
148     this._buttonsDOMElement.appendChild(this._buttons[i]);
149   }
150   this._DOMElement.appendChild(this._buttonsDOMElement);
151   this._DOMElement.appendChild(this._CSRFDOMElement);
152 };

214 /**
215  * Disables all fields in the form.
216  * @returns {sForm} The object to allow method chaining.
217  */
218 sForm.prototype.disable = function () {
219   var form = this._DOMElement;
220   var inputs = form.getElementsByTagName('input');
221   var textareas = form.getElementsByTagName('textarea');
222 
223   for (var i = 0; i < inputs.length; i++) {
224     if (inputs[i].getAttribute('type').toLowerCase() !== 'hidden') {
225       inputs[i].setAttribute('disabled', 'disabled');
226     }
227   }
228 
229   for (i = 0; i < textareas.length; i++) {
230     textareas[i].setAttribute('disabled', 'disabled');
231   }
232 
233   return this;
234 };
235 /**
236  * Enables all fields in the form.
237  * @returns {sForm} The object to allow method chaining.
238  */
239 sForm.prototype.enable = function () {
240   var form = this._DOMElement;
241   var inputs = form.getElementsByTagName('input');
242   var textareas = form.getElementsByTagName('textarea');
243 
244   for (var i = 0; i < inputs.length; i++) {
245     inputs[i].removeAttribute('disabled');
246   }
247 
248   for (i = 0; i < textareas.length; i++) {
249     textareas[i].removeAttribute('disabled');
250   }
251 
252   return this;
253 };
254 /**
255  * Submits the form via AJAX.
256  * @param {function()} cb The callback for success.
257  * @param {function()} [errorCb] The callback on error.
258  * @param {string} [dataType] The data type.
259  * @param {boolean} [isFileUpload] If the form submission is for a file upload.
260  * @returns {sForm} The object to allow method chaining.
261  */
262 sForm.prototype.submitByAJAX = function (cb, errorCb, dataType, isFileUpload) {
263   var postData = {};
264 
265   // Find anything with a value
266   var form = this._DOMElement;
267   var inputs = form.getElementsByTagName('input');
268   var textareas = form.getElementsByTagName('textarea');
269   var type;
270   var name;
271 
272   for (var i = 0; i < inputs.length; i++) {
273     type = inputs[i].getAttribute('type').toLowerCase();
274     name = inputs[i].getAttribute('name');
275 
276     if (name) {
277       if (type === 'checkbox' || type === 'radio') {
278         if (inputs[i].checked) {
279           postData[name] = inputs[i].value;
280         }
281       }
282       else {
283         postData[name] = inputs[i].value;
284       }
285     }
286   }
287 
288   for (i = 0; i < textareas.length; i++) {
289     name = textareas[i].getAttribute('name');
290 
291     if (name) {
292       postData[name] = textareas[i].value;
293     }
294   }
295 
296   sAJAXRequest.post(this._DOMElement.action, postData, cb, errorCb, dataType, isFileUpload);
297 
298   return this;
299 };
