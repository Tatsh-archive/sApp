1 /**
  2  * Represents a one-line text field.
  3  * @constructor
  4  * @param {string} name Name for the field.
  5  * @returns {sTextField} The text field object.
  6  * @augments sFormField
  7  */
  8 var sTextField = function (name) {
  9   this.parent.constructor.call(this);
 10 
 11   /**
 12    * @type (Element|null)
 13    * @private
 14    */
 15   this._labelDOMElement = null;
 16 
 17   /**
 18    * @type Element
 19    * @private
 20    */
 21   this._fieldDOMElement = document.createElement('input');
 22   this._fieldDOMElement.setAttribute('type', 'text');
 23   this._fieldDOMElement.setAttribute('name', name);
 24   this._fieldDOMElement.className = 'form-textfield';
 25 
 26   /**
 27    * @type Element
 28    * @private
 29    */
 30   this._DOMElement = document.createElement('div');
 31   this._DOMElement.className = 'form-textfield-container';
 32   this._DOMElement.appendChild(this._fieldDOMElement);
 33 
 34   return this;
 35 };
 36 /**
 37  * @type sFormField
 38  * @private
 39  */
 40 sTextField.prototype = new sFormField();
 41 sTextField.prototype.parent = sFormField.prototype;
 42 /**
 43  * Set the placeholder text of the field.
 44  * @param {string} placeholder The placeholder text.
 45  * @returns {sTextAreaField} The object to allow method chaining.
 46  */
 47 sTextField.prototype.setPlaceholder = function (placeholder) {
 48   this._fieldDOMElement.setAttribute('placeholder', placeholder);
 49   inputPlaceholder(this._fieldDOMElement);
 50   return this;
 51 };
